    export async function checkInHabit(userId: string, habitId: HabitId): Promise<{ habitId: HabitId; date: Date; checkedAt: Date }> {
      const userObjId = new mongoose.Types.ObjectId(userId);
      const dayStart = vietnamDayStart(new Date());
      const result = await HabitLog.findOneAndUpdate(
        { userId: userObjId, date: dayStart, habitId },
        { $setOnInsert: { checkedAt: new Date() } },
        { upsert: true, new: true }
      );
      return { habitId: result.habitId, date: result.date, checkedAt: result.checkedAt };
    }

    export async function getTodayHabits(userId: string): Promise<{ completed: string[]; progress: { count: number; percent: number } }> {
      const userObjId = new mongoose.Types.ObjectId(userId);
      const dayStart = vietnamDayStart(new Date());
      const logs = await HabitLog.find({ userId: userObjId, date: dayStart }).select('habitId').lean();
      const completed = logs.map(l => l.habitId);
      const count = completed.length;
      const percent = Math.round((count / 6) * 100);
      return { completed, progress: { count, percent } };
    }

    export async function getWeeklyHeatmap(userId: string): Promise<Array<{ date: string; qualified: boolean }>> {
      const userObjId = new mongoose.Types.ObjectId(userId);
      const { start } = lastNDaysRange(7);
      const agg = await HabitLog.aggregate([
        { $match: { userId: userObjId, date: { $gte: start } } },
        { $group: { _id: '$date', habits: { $addToSet: '$habitId' } } },
        { $project: { _id: 1, count: { $size: '$habits' } } },
      ]);
      // Build 7-entry array oldest → newest
      const byDate = new Map(agg.map(a => [new Date(a._id).getTime(), a.count]));
      const result: Array<{ date: string; qualified: boolean }> = [];
      const dayMs = 86400000;
      const todayStart = vietnamDayStart(new Date());
      for (let i = 6; i >= 0; i--) {
        const d = new Date(todayStart.getTime() - i * dayMs);
        const count = byDate.get(d.getTime()) ?? 0;
        result.push({ date: d.toISOString().slice(0, 10), qualified: count >= 3 });
      }
      return result;
    }

    export async function getStreak(userId: string): Promise<{ streakDays: number }> {
      // Streak rule (D-49, D-50): consecutive days where >=3 distinct habits completed (UTC+7 buckets).
      // If today has <3 (in-progress), skip today and continue from yesterday.
      // If yesterday and prior have <3, streak = (today count >= 3 ? 1 : 0).
      const userObjId = new mongoose.Types.ObjectId(userId);
      const todayStart = vietnamDayStart(new Date());
      const lookbackStart = new Date(todayStart.getTime() - 365 * 86400000);
      const agg = await HabitLog.aggregate([
        { $match: { userId: userObjId, date: { $gte: lookbackStart } } },
        { $group: { _id: '$date', habits: { $addToSet: '$habitId' } } },
        { $project: { _id: 1, count: { $size: '$habits' } } },
      ]);
      const qualifiedDays = new Set(agg.filter(a => a.count >= 3).map(a => new Date(a._id).getTime()));
      const dayMs = 86400000;
      let streak = 0;
      const todayQualified = qualifiedDays.has(todayStart.getTime());
      // If today qualifies, count it AND keep going backwards.
      // If today does not qualify, start at yesterday.
      const startOffset = todayQualified ? 0 : 1;
      for (let i = startOffset; i < 365; i++) {
        const dayTs = todayStart.getTime() - i * dayMs;
        if (qualifiedDays.has(dayTs)) streak++;
        else break;
      }
      return { streakDays: streak };
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| mobile client → /api/habits/* | Untrusted JSON body (habitId, possibly spoofed userId) |
| /api/habits/* → MongoDB | Per-user reads/writes; HabitLog unique index enforces 1-per-day |
| Aggregation → User isolation | Streak + weekly heatmap leak prevention via $match |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-14 | Elevation of Privilege | userId injection via body | mitigate | Zod `.strict()`; userId from `(req as AuthRequest).user.id` only (same pattern as T-03-07 in Plan 03-02) |
| T-03-15 | Information Disclosure | Cross-tenant streak/heatmap leak | mitigate | `$match: { userId: ObjectId(req.user.id) }` is FIRST aggregation stage in every read pipeline; HabitLog uses ObjectId on userId so string match would not work — coerce via `new mongoose.Types.ObjectId(userId)` |
| T-03-16 | Tampering | habitId NoSQL injection | mitigate | `z.enum(HABIT_IDS)` matches only 6 hard-coded strings before service touches DB |
| T-03-17 | Spoofing | Anonymous check-in | mitigate | `authenticate` on all 4 routes (verified by grep -c >= 4) |
| T-03-18 | Tampering | Duplicate check-in causes 500 from unique index violation | mitigate | `findOneAndUpdate` with `upsert: true` returns the existing doc instead of throwing duplicate-key — verified by integration test case 4 |
| T-03-19 | Denial of Service | 365-day streak loop on a user with no logs | accept | Algorithm terminates on first non-qualifying day; users with zero logs return streak=0 in O(1) since `qualifiedDays` set is empty. Worst case O(365) iterations bounded. |
| T-03-SC | Tampering | npm install of new dependencies | accept | No new dependencies — reuses Plan 03-02's `backend/src/utils/date.ts` and existing zod/mongoose/supertest stack |
</threat_model>
