    type NutritionResult = { foods: NutritionFoodItem[]; totals: { calories: number; protein: number; carbs: number; fat: number; }; aiProvider: 'logmeal' | 'openai' | 'manual'; imageUrl: string | null; }

    FoodScanState interface:
    - scanResult: NutritionResult | null
    - isScanning: boolean
    - pendingImageUri: string | null
    - setScanResult: (result: NutritionResult) => void
    - setIsScanning: (v: boolean) => void
    - setPendingImageUri: (uri: string | null) => void
    - clearScan: () => void

    Export useFoodScanStore = create<FoodScanState>((set) => ({ ...defaults, setScanResult: (result) => set({ scanResult: result, isScanning: false }), setIsScanning: (v) => set({ isScanning: v }), setPendingImageUri: (uri) => set({ pendingImageUri: uri }), clearScan: () => set({ scanResult: null, isScanning: false, pendingImageUri: null }) }))

    Imports: apiClient from './client'. Import types from './types' (IScanFoodResponse, IFoodLogItem, IFoodLog, IFoodItem).

    5 exported functions:

    scanFoodApi(imageUri: string): Promise<IScanFoodResponse>
    - Create FormData, append 'image' field as { uri: imageUri, type: 'image/jpeg', name: 'meal.jpg' } cast to any
    - Call apiClient.post('/api/food/scan', formData, { headers: { 'Content-Type': 'multipart/form-data' }, timeout: 30000 })
    - Return res.data.data

    saveFoodLogApi(body: { foods: IFoodLogItem[]; totals: { calories: number; protein: number; carbs: number; fat: number; }; aiProvider: 'openai' | 'logmeal' | 'manual'; imageUrl?: string | null; }): Promise<IFoodLog>
    - apiClient.post('/api/food/logs', body) — no timeout override (normal 10s ok)
    - Return res.data.data

    getFoodLogsApi(date: string): Promise<IFoodLog[]>
    - apiClient.get('/api/food/logs', { params: { date } })
    - Return res.data.data

    searchFoodItemsApi(q: string): Promise<IFoodItem[]>
    - apiClient.get('/api/food/items', { params: { q } })
    - Return res.data.data

    deleteFoodLogApi(logId: string): Promise<void>
    - apiClient.delete(`/api/food/logs/${logId}`)
    - Return (no data needed)
