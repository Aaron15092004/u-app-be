    ```typescript
    let isRefreshing = false;
    let refreshSubscribers: Array<(token: string | null) => void> = [];

    apiClient.interceptors.response.use(
      (r) => r,
      async (err) => {
        if (!axios.isAxiosError(err) || err.response?.status !== 401 || !err.config) return Promise.reject(err);
        // Avoid infinite loop on /refresh itself
        if (err.config.url?.includes('/api/auth/refresh')) return Promise.reject(err);
        // Avoid retrying twice
        if ((err.config as any).__isRetry) {
          // Refresh succeeded but the retried request also got 401 → escalate
          authEvents.emit('session-expired');
          return Promise.reject(err);
        }

        const original = err.config;

        if (isRefreshing) {
          // Wait for the in-flight refresh
          return new Promise((resolve, reject) => {
            refreshSubscribers.push((token) => {
              if (!token) return reject(err);
              original.headers!.Authorization = `Bearer ${token}`;
              (original as any).__isRetry = true;
              resolve(apiClient(original));
            });
          });
        }

        isRefreshing = true;
        try {
          const stored = await getRefreshToken();
          if (!stored) throw new Error('No refresh token');
          const res = await axios.post(`${apiClient.defaults.baseURL}/api/auth/refresh`, { refreshToken: stored });
          const { accessToken, refreshToken } = res.data.data;
          await saveRefreshToken(refreshToken);
          useAuthStore.setState({ accessToken });
          refreshSubscribers.forEach((cb) => cb(accessToken));
          refreshSubscribers = [];
          original.headers!.Authorization = `Bearer ${accessToken}`;
          (original as any).__isRetry = true;
          return apiClient(original);
        } catch (refreshErr) {
          refreshSubscribers.forEach((cb) => cb(null));
          refreshSubscribers = [];
          await clearTokens();
          useAuthStore.getState().clear();
          authEvents.emit('session-expired');
          return Promise.reject(refreshErr);
        } finally {
          isRefreshing = false;
        }
      }
    );
    ```

      const { isLoading, isAuthenticated, user } = useAuth();
      const segments = useSegments();
      const router = useRouter();

      useEffect(() => {
        if (isLoading) return;
        const onboardingSeen = getOnboardingSeen();
        const inAuthGroup = segments[0] === '(auth)';
        const inOnboardingGroup = segments[0] === '(onboarding)';
        const inTabsGroup = segments[0] === '(tabs)';

        if (!onboardingSeen && !inOnboardingGroup) {
          router.replace('/(onboarding)/');
        } else if (onboardingSeen && !isAuthenticated && !inAuthGroup) {
          router.replace('/(auth)/login');
        } else if (isAuthenticated && user && !user.profileCompleted && segments[1] !== 'complete-profile') {
          router.replace('/(auth)/complete-profile');
        } else if (isAuthenticated && user?.profileCompleted && (inAuthGroup || inOnboardingGroup)) {
          router.replace('/(tabs)/');
        }
      }, [isLoading, isAuthenticated, user, segments]);

      if (isLoading) return null;
      return <>{children}</>;
    }

              </RouteGate>
            </AuthProvider>
          </QueryProvider>
        </ThemeProvider>
