    import { GoogleSignin, statusCodes } from '@react-native-google-signin/google-signin';

    let configured = false;
    function configure(): void {
      if (configured) return;
      GoogleSignin.configure({
        webClientId: process.env.EXPO_PUBLIC_GOOGLE_OAUTH_CLIENT_ID_WEB!,
        // iosClientId is read from Info.plist via the expo plugin; no need here.
        offlineAccess: false,
        scopes: ['profile', 'email'],
      });
      configured = true;
    }

    export async function signInWithGoogle(): Promise<{ idToken: string }> {
      configure();
      try {
        await GoogleSignin.hasPlayServices({ showPlayServicesUpdateDialog: true });
        const userInfo = await GoogleSignin.signIn();
        const { idToken } = await GoogleSignin.getTokens();
        if (!idToken) throw new Error('No idToken returned by Google SDK');
        return { idToken };
      } catch (err: any) {
        if (err?.code === statusCodes.SIGN_IN_CANCELLED) throw new Error('CANCELLED');
        throw err;
      }
    }

    export async function signOutGoogle(): Promise<void> {
      configure();
      try { await GoogleSignin.signOut(); } catch {}
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Mobile SDK → Google identity provider | Native flow; idToken issued by Google for our bundle |
| Mobile SDK → Apple identity provider | Native flow; identityToken issued by Apple for our App ID |
| Backend → Google verifier | Audience-bound verifyIdToken — accepts ONLY our 3 client IDs |
| Backend → Apple verifier | Audience-bound verifyIdToken — accepts ONLY com.uapp.health |
| Backend → MongoDB | New users may be created from OAuth payload — name/avatar/email are attacker-controllable via spoofed providers |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-35 | Spoofing | forged Google idToken | mitigate | google-auth-library verifies signature against Google's published JWKS; audience locked to our 3 client IDs — forged tokens fail verification |
| T-02-36 | Spoofing | forged Apple identityToken | mitigate | apple-signin-auth verifies signature against Apple JWKS; audience locked to com.uapp.health |
| T-02-37 | Replay | reuse of idToken | mitigate | idTokens are short-lived (Google ~1h, Apple ~10min); JWT exp checked by verifier libraries |
| T-02-38 | Elevation of Privilege | linking different Google account to existing email-password account | mitigate | Account-link path uses the existing user; the Google sub becomes an authProvider entry. Risk: an attacker who already controls the email could pre-claim the account via OAuth — accept this risk for v1 (standard practice; email is treated as proof of identity) |
| T-02-39 | Information Disclosure | Apple's relay email exposes user's real email | accept | We store whatever Apple gives us (relay or real); Apple's design protects users; our model is unaffected |
| T-02-40 | Information Disclosure | Google profile.picture stored as avatar | accept | avatar is opt-in surface; Google CDN URL; no PII risk beyond existing email |
| T-02-41 | Denial of Service | OAuth verifier dependent on Google/Apple JWKS endpoint | accept | google-auth-library + apple-signin-auth cache JWKS internally; transient outage retries on next request |
| T-02-42 | Tampering | EXPO_PUBLIC_GOOGLE_OAUTH_CLIENT_ID_WEB bundled in mobile JS | accept | This is a public Google client ID by design; not a secret. The secret (none — Google web client has no secret in this flow) is unused; backend uses audience verification, not client secret |
| T-02-SC | Tampering | npm install of google-auth-library, apple-signin-auth, @react-native-google-signin/google-signin, expo-apple-authentication | mitigate | All four are official/well-established packages. google-auth-library is Google's official Node SDK (>10M weekly DL). apple-signin-auth is a popular maintained library (>50k weekly DL). @react-native-google-signin/google-signin is the official RN Google SDK (>200k weekly DL). expo-apple-authentication is part of the Expo SDK. Verified legitimate during planning; no blocking checkpoint required. |
