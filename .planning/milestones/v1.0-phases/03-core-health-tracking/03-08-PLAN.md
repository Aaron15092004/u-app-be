    ```typescript
    const qc = useQueryClient();
    const [rowError, setRowError] = useState<Record<IHabitId, string | null>>(
      { water: null, vegetables: null, exercise: null, sleep: null, reading: null, 'nut-milk': null }
    );

    const todayQuery = useQuery({ queryKey: ['habits', 'today'], queryFn: getTodayHabitsApi });
    const weeklyQuery = useQuery({ queryKey: ['habits', 'weekly'], queryFn: getWeeklyHabitsApi });
    const streakQuery = useQuery({ queryKey: ['habits', 'streak'], queryFn: getStreakApi });

    const mutation = useMutation({
      mutationFn: (habitId: IHabitId) => checkInHabitApi(habitId),
      onMutate: async (habitId) => {
        await qc.cancelQueries({ queryKey: ['habits', 'today'] });
        const previous = qc.getQueryData<ITodayHabits>(['habits', 'today']);
        qc.setQueryData<ITodayHabits>(['habits', 'today'], (old) => {
          if (!old) return old;
          const completed = old.completed.includes(habitId) ? old.completed : [...old.completed, habitId];
          return { completed, progress: { count: completed.length, percent: Math.round((completed.length / 6) * 100) } };
        });
        setRowError((e) => ({ ...e, [habitId]: null }));
        return { previous, habitId };
      },
      onError: (_err, habitId, ctx) => {
        if (ctx?.previous) qc.setQueryData(['habits', 'today'], ctx.previous);
        setRowError((e) => ({ ...e, [habitId]: 'Không thể cập nhật. Thử lại.' }));
      },
      onSettled: () => {
        qc.invalidateQueries({ queryKey: ['habits', 'today'] });
        qc.invalidateQueries({ queryKey: ['habits', 'weekly'] });
        qc.invalidateQueries({ queryKey: ['habits', 'streak'] });
      },
    });

    const completedSet = new Set(todayQuery.data?.completed ?? []);
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Mobile → Backend /api/habits/* | Bearer token via apiClient; userId from JWT on backend (T-03-14 mitigated) |
| Optimistic update → cache | Mutates queryClient state pre-server-confirm; rollback on error |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-40 | Tampering | Client-side optimistic check-in claims completion that never persisted | mitigate | onError rollback restores prior `completed` array; onSettled invalidates the today query so the next refresh always reflects server truth |
| T-03-41 | Information Disclosure | Habit data from a previous user surviving logout | mitigate | AuthProvider.logout() calls `queryClient.clear()` (Phase 2); local rowError state is component-scoped and resets on unmount |
| T-03-42 | Tampering | Hardcoded `HABIT_DEFAULTS` drifting from backend enum | mitigate | Both mobile and backend declare the 6 ID strings; integration test in Plan 03-03 (`Thói quen không hợp lệ` error) protects against drift — any new habit ID on mobile would 400 |
| T-03-43 | Spoofing | Anonymous check-in | mitigate | Backend `authenticate` on all 4 habit routes (Plan 03-03 grep gate) |
| T-03-44 | Denial of Service | Tap-spamming "Đánh dấu +1" generating 100 POSTs | accept | `mutation.isPending` disables the row during inflight; backend upsert is idempotent (Plan 03-03 T-03-18); even if 100 requests landed, DB writes only 1 row. |
| T-03-SC | Tampering | npm install of new dependencies | accept | No new dependencies — `@tanstack/react-query` already installed |
</threat_model>
